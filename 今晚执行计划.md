```powershell
docker volume ls | grep ralph-backup
docker run --rm -v "$(pwd):/workspace" -v "ralph-backup-XXXXXX:/backup" alpine:latest sh -c "rm -rf /workspace/* && tar xzf /backup/project-snapshot.tar.gz -C /workspace"

# 恢复 git 分支
git checkout <原始分支>
git branch -D safe-ralph-XXXXXX

# 清理 sbx
sbx rm --all --force
```
